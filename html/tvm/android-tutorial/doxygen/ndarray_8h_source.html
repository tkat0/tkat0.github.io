377 inline bool SaveDLTensor(dmlc::Stream* strm,
378  DLTensor* tensor) {
379  uint64_t header = kTVMNDArrayMagic, reserved = 0;
380  strm->Write(header);
381  strm->Write(reserved);
382  // Always save data as CPU context
383  //
384  // Parameters that get serialized should be in CPU by default.
385  // So even the array's context is GPU, it will be stored as CPU array.
386  // This is used to prevent case when another user loads the parameters
387  // back on machine that do not have GPU or related context.
388  //
389  // We can always do array.CopyTo(target_ctx) to get a corresponding
390  // array in the target context.
391  DLContext cpu_ctx;
392  cpu_ctx.device_type = kDLCPU;
393  cpu_ctx.device_id = 0;
394  strm->Write(cpu_ctx);
395  strm->Write(tensor->ndim);
396  strm->Write(tensor->dtype);
397  int ndim = tensor->ndim;
398  strm->WriteArray(tensor->shape, ndim);
399  int type_bytes = tensor->dtype.bits / 8;
400  int64_t num_elems = 1;
401  for (int i = 0; i < ndim; ++i) {
402  num_elems *= tensor->shape[i];
403  }
404  int64_t data_byte_size = type_bytes * num_elems;
405  strm->Write(data_byte_size);
406 
407  if (DMLC_IO_NO_ENDIAN_SWAP &&
408  tensor->ctx.device_type == kDLCPU &&
409  tensor->strides == nullptr &&
410  tensor->byte_offset == 0) {
411  // quick path
412  strm->Write(tensor->data, data_byte_size);
413  } else {
414  std::vector<uint8_t> bytes(data_byte_size);
415  CHECK_EQ(TVMArrayCopyToBytes(
416  tensor, dmlc::BeginPtr(bytes), data_byte_size), 0)
417  << TVMGetLastError();
418  if (!DMLC_IO_NO_ENDIAN_SWAP) {
419  dmlc::ByteSwap(dmlc::BeginPtr(bytes), type_bytes, num_elems);
420  }
421  strm->Write(dmlc::BeginPtr(bytes), data_byte_size);
422  }
423  return true;
424 }

430 inline bool NDArray::Load(dmlc::Stream* strm) {
431  uint64_t header, reserved;
432  CHECK(strm->Read(&header))
433  << "Invalid DLTensor file format";
434  CHECK(strm->Read(&reserved))
435  << "Invalid DLTensor file format";
436  CHECK(header == kTVMNDArrayMagic)
437  << "Invalid DLTensor file format";
438  DLContext ctx;
439  int ndim;
440  DLDataType dtype;
441  CHECK(strm->Read(&ctx))
442  << "Invalid DLTensor file format";
443  CHECK(strm->Read(&ndim))
444  << "Invalid DLTensor file format";
445  CHECK(strm->Read(&dtype))
446  << "Invalid DLTensor file format";
447  CHECK_EQ(ctx.device_type, kDLCPU)
448  << "Invalid DLTensor context: can only save as CPU tensor";
449  std::vector<int64_t> shape(ndim);
450  if (ndim != 0) {
451  CHECK(strm->ReadArray(&shape[0], ndim))
452  << "Invalid DLTensor file format";
453  }
454  NDArray ret = NDArray::Empty(shape, dtype, ctx);
455  int64_t num_elems = 1;
456  int elem_bytes = (ret->dtype.bits + 7) / 8;
457  for (int i = 0; i < ret->ndim; ++i) {
458  num_elems *= ret->shape[i];
459  }
460  int64_t data_byte_size;
461  CHECK(strm->Read(&data_byte_size))
462  << "Invalid DLTensor file format";
463  CHECK(data_byte_size == num_elems * elem_bytes)
464  << "Invalid DLTensor file format";
465  CHECK(strm->Read(ret->data, data_byte_size))
466  << "Invalid DLTensor file format";
467  if (!DMLC_IO_NO_ENDIAN_SWAP) {
468  dmlc::ByteSwap(ret->data, elem_bytes, num_elems);
469  }
470  *this = ret;
471  return true;
472 }
