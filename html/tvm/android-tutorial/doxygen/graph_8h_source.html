255 template <typename GNode, typename HashType,
256  typename FVisit, typename HashFunc,
257  typename InDegree, typename GetInput>
258 void PostOrderDFSVisit(const std::vector<GNode>& heads,
259  FVisit fvisit,
260  HashFunc hash,
261  InDegree indegree,
262  GetInput getinput) {
263  std::vector<std::pair<GNode, uint32_t> > stack;
264  std::unordered_set<HashType> visited;
265  for (auto& head : heads) {
266  HashType head_hash = hash(head);
267  if (visited.count(head_hash) == 0) {
268  stack.push_back(std::make_pair(head, 0));
269  visited.insert(head_hash);
270  }
271  while (!stack.empty()) {
272  std::pair<GNode, uint32_t>& back = stack.back();
273  if (back.second == indegree(back.first)) {
274  fvisit(back.first);
275  stack.pop_back();
276  } else {
277  const GNode& input = getinput(back.first, back.second++);
278  HashType input_hash = hash(input);
279  if (visited.count(input_hash) == 0) {
280  stack.push_back(std::make_pair(input, 0));
281  visited.insert(input_hash);
282  }
283  }
284  }
285  }

289 inline void DFSVisit(const std::vector<NodeEntry>& heads,
290  FVisit fvisit) {
291  typedef const NodePtr* GNode;
292  std::vector<GNode> head_nodes(heads.size());
293  std::transform(heads.begin(), heads.end(), head_nodes.begin(),
294  [](const NodeEntry& e)->GNode {
295  return &e.node;
296  });
297  PostOrderDFSVisit<GNode, Node*>(
298  head_nodes,
299  [fvisit](GNode n) { fvisit(*n); }, // FVisit
300  [](GNode n)->Node* { return n->get(); }, // HashFunc
301  [](GNode n)->uint32_t { // InDegree
302  if (!(*n)) return 0;
303  return (*n)->inputs.size() + (*n)->control_deps.size();
304  },
305  [](GNode n, uint32_t index)->GNode { // GetInput
306  if (index < (*n)->inputs.size()) {
307  return &(*n)->inputs.at(index).node;
308  } else {
309  return &(*n)->control_deps.at(index - (*n)->inputs.size());
310  }
311  });
