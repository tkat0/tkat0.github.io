748  void InitByPackedArgs(const runtime::TVMArgs& args, bool allow_unknown) final {
749  CHECK_EQ(args.size() % 2, 0);
750  const int kLinearSearchBound = 16;
751  int hit_count = 0;
752  // applies two stratgies to lookup
753  if (args.size() < kLinearSearchBound) {
754  // linear search.
755  auto ffind = [&args](const char* key, runtime::TVMArgValue* val) {
756  for (int i = 0; i < args.size(); i += 2) {
757  CHECK_EQ(args.type_codes[i], kStr);
758  if (!std::strcmp(key, args.values[i].v_str)) {
759  *val = args[i + 1];
760  return true;
761  }
762  }
763  return false;
764  };
765  auto vis = ::tvm::detail::CreateInitVisitor(DerivedType::_type_key, ffind);
766  self()->__VisitAttrs__(vis);
767  hit_count = vis.hit_count_;
768  } else {
769  // construct a map then do lookup.
770  std::unordered_map<std::string, runtime::TVMArgValue> kwargs;
771  for (int i = 0; i < args.size(); i += 2) {
772  CHECK_EQ(args.type_codes[i], kStr);
773  kwargs[args[i].operator std::string()] = args[i + 1];
774  }
775  auto ffind = [&kwargs](const char *key, runtime::TVMArgValue* val) {
776  auto it = kwargs.find(key);
777  if (it != kwargs.end()) {
778  *val = it->second;
779  return true;
780  }
781  return false;
782  };
783  auto vis = ::tvm::detail::CreateInitVisitor(DerivedType::_type_key, ffind);
784  self()->__VisitAttrs__(vis);
785  hit_count = vis.hit_count_;
786  }
787  // error handling, slow path
788  if (hit_count * 2 != args.size() && !allow_unknown) {
789  for (int i = 0; i < args.size(); i += 2) {
790  ::tvm::detail::AttrExistVisitor visitor;
791  visitor.key_ = args[i].operator std::string();
792  self()->__VisitAttrs__(visitor);
793  if (!visitor.exist_) {
794  std::ostringstream os;
795  os << DerivedType::_type_key
796  << ": does not have field \'" << visitor.key_
797  << "\', Possible fields:\n";
798  os << "----------------\n";
799  this->PrintDocString(os);
800  throw AttrError(os.str());
801  }
802  }
803  }
804  }
