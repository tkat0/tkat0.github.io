45 inline std::vector<int> GetRealAxis(int ndim, const Array<Integer>& axis) {
46  std::vector<int> real_axis;
47  if (!axis.defined() || axis.size() == 0) {
48  for (int i = 0; i < ndim; ++i) {
49  real_axis.push_back(i);
50  }
51  } else {
52  // Use a set so duplicates are removed and the dims are sorted
53  for (auto elem : axis) {
54  int64_t val = elem->value;
55  if (val < 0) {
56  val += ndim;
57  }
58  CHECK_LE(val, ndim) << " exceeds the maximum dimension " << ndim;
59  CHECK_GE(val, 0);
60  real_axis.push_back(static_cast<int>(val));
61  }
62  std::sort(real_axis.begin(), real_axis.end());
63  real_axis.resize(
64  std::unique(real_axis.begin(), real_axis.end()) - real_axis.begin());
65  }
66  return real_axis;
67 }
