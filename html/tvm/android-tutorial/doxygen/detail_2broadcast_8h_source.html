27 inline BroadcastHelper BroadcastShape(const tvm::Array<tvm::Expr>& shape1,
28  const tvm::Array<tvm::Expr>& shape2) {
29  BroadcastHelper bh;
30  int s1_size = shape1.size();
31  int s2_size = shape2.size();
32  tvm::Expr one(1);
33  int i;
34  for (i = 1; i <= std::min(s1_size, s2_size); ++i) {
35  bh.all_vars.push_front(tvm::Var());
36  if (topi::detail::EqualCheck(shape1[s1_size - i], shape2[s2_size - i])) {
37  bh.common_shape.push_front(shape1[s1_size - i]);
38  bh.vars1.push_front(bh.all_vars[0]);
39  bh.vars2.push_front(bh.all_vars[0]);
40  } else if (topi::detail::EqualCheck(one, shape1[s1_size - i])) {
41  CHECK(!topi::detail::EqualCheck(one, shape2[s2_size - i]));
42  bh.common_shape.push_front(shape2[s2_size - i]);
43  bh.vars2.push_front(bh.all_vars[0]);
44  } else if (topi::detail::EqualCheck(one, shape2[s2_size - i])) {
45  bh.common_shape.push_front(shape1[s1_size - i]);
46  bh.vars1.push_front(bh.all_vars[0]);
47  } else {
48  CHECK(false) << "Incompatible broadcast dims: " << shape1[s1_size - i]
49  << " and " << shape2[s2_size - i] << " in: "
50  << tvm::Array<tvm::Expr>(shape1.begin(), shape1.end())
51  << " and "
52  << tvm::Array<tvm::Expr>(shape2.begin(), shape2.end());
53  }
54  }
55  // Remaining dimensions whether on shape1 or shape2 can always be completed
56  auto max_size = std::max(s1_size, s2_size);
57  auto& shape = (s1_size > s2_size) ? shape1 : shape2;
58  auto& vars = (s1_size > s2_size) ? bh.vars1 : bh.vars2;
59  for (; i <= max_size; ++i) {
60  bh.all_vars.push_front(tvm::Var());
61  bh.common_shape.push_front(shape[max_size - i]);
62  vars.push_front(bh.all_vars[0]);
63  }
64  return bh;
65 }

67 inline tvm::Array<tvm::Expr> InputIndexFromBroadcast(
68  const tvm::Array<tvm::Var>& ovars, const tvm::Tensor& T,
69  const std::deque<tvm::Var>& my_vars, const std::deque<tvm::Var>& all_vars) {
70  tvm::Array<tvm::Expr> ivars;
71  CHECK_EQ(ovars.size(), all_vars.size());
72  // N^2, could use a map but NBD..
73  size_t expected_dims = T->shape.size();
74  for (size_t i = 0; i < ovars.size(); ++i) {
75  bool found = false;
76  for (size_t j = 0; j < my_vars.size(); ++j) {
77  if (all_vars[i].same_as(my_vars[j])) {
78  ivars.push_back(ovars[i]);
79  found = true;
80  break;
81  }
82  }
83  // Only inject 0 here if we have not yet reached the dimension of I
84  // (i.e. this must be a 1)

86  ivars.push_back(tvm::make_zero(ovars[i].type()));
87  }
88  }
89  CHECK(expected_dims == ivars.size());
90  return ivars;
