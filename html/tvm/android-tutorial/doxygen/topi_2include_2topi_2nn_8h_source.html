151 inline tvm::Tensor pad(const tvm::Tensor& t,
152  const tvm::Array<tvm::Expr>& pad_before,
153  tvm::Array<tvm::Expr> pad_after = tvm::Array<tvm::Expr>(),
154  Expr pad_value = Expr(),
155  std::string name = "tensor",
156  std::string tag = kElementWise) {
157  if (pad_after.size() < pad_before.size()) {
158  for (size_t i = pad_after.size(); i < pad_before.size(); ++i) {
159  pad_after.push_back(pad_before[i]);
160  }
161  }
162  CHECK_GE(pad_before.size(), 1);
163  CHECK_EQ(pad_before.size(), pad_after.size());
164  tvm::Array<tvm::Expr> output_shape;
165  for (size_t i = 0; i < t->shape.size(); ++i) {
166  if (i >= pad_before.size()) {
167  output_shape.push_back(t->shape[i]);
168  } else {
169  output_shape.push_back(
170  tvm::ir::Simplify(t->shape[i] + pad_before[i] + pad_after[i]));
171  }
172  }
173 
174  if (!pad_value.defined()) {
175  pad_value = tvm::make_const(t->dtype, 0);
176  }
177 
178  auto l = [&](tvm::Array<tvm::Var> ovars) {
179  tvm::Array<tvm::Expr> indices;
180  tvm::Array<tvm::Expr> sel;
181  for (size_t i = 0; i < t->shape.size(); ++i) {
182  if (i >= pad_before.size()) {
183  indices.push_back(ovars[i]);
184  continue;
185  }
186  if (!topi::detail::EqualCheck(pad_before[i], 0)) {
187  sel.push_back(ovars[i] >= pad_before[i]);
188  indices.push_back(ovars[i] - pad_before[i]);
189  } else {
190  indices.push_back(ovars[i]);
191  }
192  if (!topi::detail::EqualCheck(pad_after[i], 0)) {
193  sel.push_back(tvm::ir::Simplify(ovars[i] < pad_before[i] + t->shape[i]));
194  }
195  }
196  if (sel.size() != 0) {
197  return tvm::if_then_else(
198  detail::Map(sel, tvm::ir::And::make), t(indices), pad_value);
199  }
200  return t(indices);
201  };
202  return tvm::compute(output_shape, l, name, tag);
203 }
