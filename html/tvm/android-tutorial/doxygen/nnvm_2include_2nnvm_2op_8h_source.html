438  // update the attribute map of the key by creating new empty if needed.
439  UpdateAttrMap(attr_name,
440  [this, attr_name, value, plevel](any* pmap) {
441  // the callback is in lockscope so is threadsafe.
442  if (pmap->empty()) {
443  OpMap<ValueType> pm;
444  pm.attr_name_ = attr_name;
445  *pmap = std::move(pm);
446  }
447  CHECK(pmap->type() == typeid(OpMap<ValueType>))
448  << "Attribute " << attr_name
449  << " of operator " << this->name
450  << " is registered as inconsistent types"
