145 inline bool find_height_width(const std::string& layout,
146  int* height_axis,
147  int* width_axis) {
148  *height_axis = -1, *width_axis = -1;
149  int curr_idx = 0;
150  for (size_t i = 0; i < layout.size(); ++i) {
151  if ((layout[i] >= 'A' && layout[i] <= 'Z') ||
152  (layout[i] >= 'a' && layout[i] <= 'z')) {
153  if (layout[i] == 'H') {
154  if (*height_axis != -1) return false;
155  *height_axis = curr_idx;
156  } else if (layout[i] == 'W') {
157  if (*width_axis != -1) return false;
158  *width_axis = curr_idx;
159  } else if (layout[i] == 'h' || layout[i] == 'w') {
160  // do not support split on height or width, e.g., NCHW16w
161  return false;
162  }
163  ++curr_idx;
164  }
165  }
166  if (*height_axis == -1 || *width_axis == -1) return false;
167  return true;
168 }
