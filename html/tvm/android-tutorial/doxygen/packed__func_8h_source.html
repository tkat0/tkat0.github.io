930 inline TVMType String2TVMType(std::string s) {
931  TVMType t;
932  // handle None type
933  if (s.length() == 0) {
934  t.bits = 0; t.lanes = 0; t.code = kHandle;
935  return t;
936  }
937  t.bits = 32; t.lanes = 1;
938  const char* scan;
939  if (s.substr(0, 3) == "int") {
940  t.code = kDLInt; scan = s.c_str() + 3;
941  } else if (s.substr(0, 4) == "uint") {
942  t.code = kDLUInt; scan = s.c_str() + 4;
943  } else if (s.substr(0, 5) == "float") {
944  t.code = kDLFloat; scan = s.c_str() + 5;
945  } else if (s.substr(0, 6) == "handle") {
946  t.code = kHandle;
947  t.bits = 64; // handle uses 64 bit by default.
948  scan = s.c_str() + 6;
949  } else if (s == "bool") {
950  t.code = kDLUInt;
951  t.bits = 1;
952  t.lanes = 1;
