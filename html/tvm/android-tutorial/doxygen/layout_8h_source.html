403  void parse(const std::string& layout) {
404  name_ = layout;
405  std::fill_n(superdim_pos_, kUniqueDim, -1);
406  std::fill_n(subdim_pos_, kUniqueDim, -1);
407  std::fill_n(subdim_size_, kUniqueDim, -1);
408  layout_simplified_.clear();
409 
410  if (layout == "__undef__") return;
411 
412  int32_t factor = 0;
413  uint32_t curr = 0;
414  for (size_t i = 0; i < layout.size(); ++i) {
415  const LayoutDim c = layout.at(i);
416  if (is_superdim(c)) {
417  int pos = c - 'A';
418  CHECK_EQ(factor, 0) << "Invalid layout " << layout
419  << ": invalid factor size " << factor
420  << " before dimension " << c;
421  CHECK_EQ(superdim_pos_[pos], -1) << "Invalid layout " << layout
422  << ": duplicate dimension " << c;
423  superdim_pos_[pos] = curr++;
424  layout_simplified_.push_back(c);
425  } else if (is_subdim(c)) {
426  int pos = c - 'a';
427  CHECK_GT(factor, 0) << "Invalid layout " << layout << ": invalid factor size "
428  << factor << " for dimension " << c;
429  CHECK_EQ(subdim_pos_[pos], -1) << "Invalid layout " << layout

431  CHECK_EQ(subdim_size_[pos], -1) << "Invalid layout " << layout
432  << ": duplicate dimension " << c;
433  subdim_pos_[pos] = curr++;
434  subdim_size_[pos] = factor;
435  layout_simplified_.push_back(c);
436  factor = 0;
437  } else if (c >= '0' && c <= '9') {
438  CHECK(factor >= 0) << "Invalid layout " << layout << ": _ is adjacent to a number.";
